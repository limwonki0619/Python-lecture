def Normal_create(start, end, interval, mu, sd, custom):
    xx = np.linspace(start, end, interval)
    rv = sp.stats.norm(mu, sd)
    pdf = rv.pdf(xx)
    plt.plot(xx, pdf, custom)
    plt.title("mu = {0} sd = {1} 정규분포 확률밀도함수".format(mu, sd))
    plt.xlabel("$x$")
    plt.ylabel("$p(x)$")
    plt.ylim(0,1)
    plt.show()
    
# plot1    
Normal_create(-3, 3, 600, 0, 0.45, 'r--')

# plot2
Normal_create(-3, 3, 600, 0, 0.75, 'yo-.')

# plot3
Normal_create(-3, 3, 600, 0, 1, 'b2--')
